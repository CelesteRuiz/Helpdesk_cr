Clase: Ticket

Atributos:
- id: int
- descripcion: string
- tipo: string (Hardware, Red, Software)
- estado: string (Abierto, En proceso, Resuelto, Cerrado)
- fechaCreacion: date
- tecnicoAsignado: Tecnico

Métodos:
- crearTicket(descripcion, tipo)
- asignarTecnico(tecnico)
- cambiarEstado(nuevoEstado)
- obtenerDetalles()
